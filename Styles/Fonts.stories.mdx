# Fonts
Xfinity Font scale and font creation for TV platform

## Source

https://github.comcast.com/Lightning/lightning-ui/packages/@lightning/ui-core/src/components/Fonts.stories/Fonts.stories.js

## Usage

```js
import { createFonts, XfinityFonts } from '@lightning/ui/Styles';

class App extends lng.Application {
  static getFonts() {
    return createFonts(XfinityFonts)
  }
}
```

### Multiple Fonts

```js
export const CoolFontFace = {
  default: {
    fontName: 'MyCoolFont',
    baseUrl:
      '/assets/fonts/MyCoolFont/',
    weights: {
      regular: {
        family: 'Regular',
        url: 'mycoolfont.woff'
      },
      bold: {
        family: 'Bold',
        url: 'mycoolfont-bolc.woff'
      },
  },
};

class App extends lng.Application {
  static getFonts() {
    return createFonts(XfinityFonts, CoolFontFace)
  }
}
```
